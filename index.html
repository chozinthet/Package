Greet Function

    HTML
    <h1 id="forTime"></h1>

    JAVASCRIPT
    greet();
    function greet() {
        var d1 = new Date();
        var hour = d1.getHours();
        if(hour < 11){
            document.getElementById("forTime").innerHTML = "Good Morning! CHO ";
        }else{
            if(hour < 16){
                document.getElementById("forTime").innerHTML = "Good Afternoon! CHO ";
            }
            else{
                if (hour < 19) {
                    document.getElementById("forTime").innerHTML = "Good Evening! CHO ";
                } else {
                    if (hour <= 24) {
                        document.getElementById("forTime").innerHTML = "Good Night! CHO ";
                    }
                }
            }
        }
    }
